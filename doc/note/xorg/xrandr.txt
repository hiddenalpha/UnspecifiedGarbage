## Grab moving names

LG4K_TUXSIX=$(xrandr | grep ' connected' | grep 'HDMI-' | cut -d' ' -f1)
POST_TUXSIX=$(xrandr | grep ' connected' | grep 'DP-' | cut -d' ' -f1)


## Setup PostLeftPortraitPlusLg4kRight

xrandr --fb "$((1080+2560))x1920" \
    --delmonitor post --delmonitor lg4k \
    --setmonitor post "1080/324x1920/520+0+0" "${POST:?}" \
    --setmonitor lg4k "2560/700x1440/393+1080+350" "${LG4K:?}" \
    --output "${POST:?}" --rotate left --panning 0x0 \
    ;
xrandr --delmonitor post --delmonitor lg4k
