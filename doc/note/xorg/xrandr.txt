

## WARN Quirks Ahead!

ChatGPT recommends this command order to prevent conflicts while
applying.

  mode, rate, dpi, scale, rotate, pos, panning, primary.

Do NOT combine commands in one invocation! It will just randomly end up
in trouble and just ignore some of the settings. DON'T DO IT!



## Grab moving names

LG4K_TUXSIX=$(xrandr | grep ' connected' | grep 'HDMI-' | cut -d' ' -f1)
POST_TUXSIX=$(xrandr | grep ' connected' | grep 'DP-' | cut -d' ' -f1)


## LG 4k @ tux-six
# LG4K_DPI := 2560 / (700 / 25.4)
true \
  && xrandr --output "{LG4K_TUXSIX:?}" --mode 2560x1440 \
  && xrandr --output "{LG4K_TUXSIX:?}" --rate 60.00 \
  && xrandr --output "{LG4K_TUXSIX:?}" --dpi 93 \
  && xrandr --output "{LG4K_TUXSIX:?}" --pos 1200x3 \
  && xrandr --output "{LG4K_TUXSIX:?}" --primary \


## Post HD @ tux-six
# POST_DPI := 1920 / (518 / 25.4)
true \
  && xrandr --output "${POST_TUXSIX:?}" --mode 1920x1200 \
  && xrandr --output "${POST_TUXSIX:?}" --rate 59.95 \
  && xrandr --output "${POST_TUXSIX:?}" --dpi 94 \
  && xrandr --output "${POST_TUXSIX:?}" --rotate left \
  && xrandr --output "${POST_TUXSIX:?}" --pos 0x0 \
  && xrandr --output "${POST_TUXSIX:?}" --primary \
